        function toggleFaq(id) {

        // Dynamic Countdown Timer to June 30th of the current year (or next year if past June)
        function initCountdown() {
            let now = new Date();
            let currentYear = now.getFullYear();
            let targetDate = new Date(`June 30, ${currentYear} 23:59:59`);
            
            // If we are already past June 30 of the current year, target the next year
            if (now > targetDate) {
                targetDate = new Date(`June 30, ${currentYear + 1} 23:59:59`);
            }

            function updateTimer() {
                const nowMs = new Date().getTime();
                const diff = targetDate.getTime() - nowMs;

                if (diff <= 0) {
                    document.getElementById('countdown-timer').innerHTML = "Promo Scaduta!";
                    return;
                }

                const days = Math.floor(diff / (1000 * 60 * 60 * 24));
                const hours = Math.floor((diff % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
                const minutes = Math.floor((diff % (1000 * 60 * 60)) / (1000 * 60));
                const seconds = Math.floor((diff % (1000 * 60)) / 1000);

                // Main countdown elements
                const daysElem = document.getElementById('days');
                const hoursElem = document.getElementById('hours');
                const minsElem = document.getElementById('minutes');
                const secsElem = document.getElementById('seconds');

                if (daysElem) daysElem.textContent = String(days).padStart(2, '0');
                if (hoursElem) hoursElem.textContent = String(hours).padStart(2, '0');
                if (minsElem) minsElem.textContent = String(minutes).padStart(2, '0');
                if (secsElem) secsElem.textContent = String(seconds).padStart(2, '0');

                // Top banner countdown format
                const bannerCount = document.getElementById('countdown-banner');
                if (bannerCount) {
                    bannerCount.textContent = `Scade in: ${days}g ${hours}o ${minutes}m ${seconds}s`;
                }
            }

            setInterval(updateTimer, 1000);
            updateTimer();
        }

        initCountdown();
